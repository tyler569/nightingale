cmake_minimum_required(VERSION 3.10)

list(APPEND CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}/toolchain/CMake")

set(CMAKE_C_STANDARD 23)
set(CMAKE_C_FLAGS_DEBUG "-g -O0 -Werror")
set(CMAKE_C_FLAGS_RELEASE "-O3")
set(CMAKE_EXE_LINKER_FLAGS_DEBUG "-g")

set(autogen_include_dir "${CMAKE_BINARY_DIR}/autogenerated" CACHE PATH "Directory for autogenerated headers")

file(MAKE_DIRECTORY ${autogen_include_dir})

message(STATUS "CMAKE_BUILD_TYPE: ${CMAKE_BUILD_TYPE}")

project(nightingale C ASM)

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)
# include_directories(${CMAKE_SOURCE_DIR}/include)
# Configure version.h for this target
configure_file(${CMAKE_SOURCE_DIR}/../include/version.h.in ${CMAKE_CURRENT_BINARY_DIR}/version.h)
# Sysroot headers are handled by the toolchain file

add_compile_options(
    -fno-asynchronous-unwind-tables
    -fno-omit-frame-pointer
    -fsanitize=undefined
)

add_compile_definitions(__kernel__=1 _NG_SOURCE=1)

# Modules are built separately by nightingale-modules package
# add_subdirectory(modules)

add_executable(nightingale_kernel
    ../shared/libc/ctype.c
    ../shared/libc/errno.c
    ../shared/libc/hexdump.c
    ../shared/libc/malloc.c
    ../shared/libc/print.c
    ../shared/libc/qsort.c
    ../shared/libc/rbtree.c
    ../shared/libc/signal.c
    ../shared/libc/stat.c
    ../shared/libc/stdlib.c
    ../shared/libc/stream.c
    ../shared/libc/stream_ring.c
    ../shared/libc/string.c
    ../shared/libc/timeconv.c
    ../shared/linker/elf-ng.c
    ../shared/linker/modld.c
    drv/e1000.c
    drv/irq.c
    drv/pci.c
    drv/rtl8139.c
    fs/char_dev.c
    fs/dentry.c
    fs/file.c
    fs/file_system.c
    fs/inet_socket.c
    fs/init.c
    fs/initfs.c
    fs/pipe.c
    fs/proc.c
    fs/proc_files.c
    fs/syscalls.c
    fs/tarfs.c
    fs/tmpfs.c
    fs/tty.c
    fs/unix_socket.c
    fs/vnode.c
    lib/chacha20.c
    lib/commandline.c
    lib/debug.c
    lib/event_log.c
    lib/font.c
    lib/limine.c
    lib/mod.c
    lib/print_test.c
    lib/random.c
    lib/string.c
    lib/trace.c
    lib/ubsan.c
    main.c
    lib/dmgr.c
    mem/mman.c
    mem/pmm.c
    mem/ringbuf.c
    mem/spalloc.c
    mt/elf.c
    mt/exec.c
    mt/mutex.c
    mt/signal.c
    mt/spin.c
    mt/sync_testbed.c
    mt/thread.c
    net/debug.c
    net/filter.c
    net/pk.c
    net/udp.c
    net/utils.c
    syscall.c
    tests.c
    time.c
    timer.c
    tty.c
    uname.c
    video.c
    net/ip_ingress.c
    net/arp.c
    net/net_ingress.c
    net/worker.c
    net/ip_egress.c
)

# Add binary directory to include path for generated headers
target_include_directories(nightingale_kernel PRIVATE ${CMAKE_CURRENT_BINARY_DIR})

if (${CMAKE_SYSTEM_PROCESSOR} STREQUAL "X86_64")
    target_sources(nightingale_kernel PRIVATE
        arch/x86_64/acpi.c
        arch/x86_64/arch.c
        arch/x86_64/ata.c
        arch/x86_64/backtrace.c
        arch/x86_64/boot.S
        arch/x86_64/cpu.c
        arch/x86_64/gdt.c
        arch/x86_64/halt.c
        arch/x86_64/interrupt.c
        arch/x86_64/ioapic.c
        arch/x86_64/isrs.S
        arch/x86_64/lapic.c
        arch/x86_64/pic.c
        arch/x86_64/pit.c
        arch/x86_64/rtc.c
        arch/x86_64/serial.c
        arch/x86_64/uart.c
        arch/x86_64/vmm.c
        ../libc/x86_64/setjmp.S
        ../libc/x86_64/nightingale.c
    )

    target_compile_options(nightingale_kernel PRIVATE
        -mno-red-zone
        -mno-mmx
        -mno-sse
        -mno-sse2
        -mcmodel=kernel
    )

    set(NG_KERNEL_LINK_SCRIPT "${CMAKE_CURRENT_SOURCE_DIR}/arch/x86_64/link_hh.ld")
elseif (${CMAKE_SYSTEM_PROCESSOR} STREQUAL "aarch64")
    target_sources(nightingale_kernel PRIVATE
    )

    target_compile_options(nightingale_kernel PRIVATE
        -mgeneral-regs-only
    )

    set(NG_KERNEL_LINK_SCRIPT "${CMAKE_CURRENT_SOURCE_DIR}/arch/aarch64/link_hh.ld")
else ()
    message(FATAL_ERROR "Unsupported architecture: ${CMAKE_SYSTEM_PROCESSOR}")
endif ()

set_target_properties(nightingale_kernel PROPERTIES
    LINK_DEPENDS "${NG_KERNEL_LINK_SCRIPT}"
)

# Install targets
install(TARGETS nightingale_kernel
    RUNTIME DESTINATION boot
    COMPONENT kernel)

# Install kernel headers for modules
install(DIRECTORY include/
    DESTINATION include
    COMPONENT kernel-headers
    FILES_MATCHING PATTERN "*.h")

set(CMAKE_C_LINK_FLAGS " \
    -T${NG_KERNEL_LINK_SCRIPT} \
    -z max-page-size=0x1000 \
")
